    sentence = sentence.lower()
    sentence = re.sub(r"[^A-Za-z0-9]", " ", sentence)
    tokens = word_tokenize(sentence)
    tokens = [token for token in tokens if token not in stopwords]

    return tokens

def build_vocab(docs):
    vocab = []
    for doc in docs:
        for token in doc:
            if token not in vocab:
                vocab.append(token)

    return vocab

def tokens_sequencing(vocab, docs):
    sequenceColumn = []
    for doc in docs:
        sequence = []
        for token in doc:
            if token in vocab:
                sequence.append(vocab.index(token))
        sequenceColumn.append(sequence)
    
    return sequenceColumn

1.2 Complétez la classe SkipGramDataLoader qui génère des exemples d'entraînement.¶

L'objectif du skipgram est de prendre en entre un mot quelconque et de prdire un mot faisant partie de son entourage (l'entourage tant dtermin par la taille de la fentre (window size)). Sois le jeton $w(t)$ la position $t$ avec une fentre de mots de taille 5 (incluant le mot cible), le modle doit prdire les jetons $w(t-2)$, $w(t-1)$, $w(t+1)$, $w(t+2)$. Ici votre skip-gram a une taille k de 2.

Pour ce faire nous crons des paires de mots arranges de la manire suivante: jeton d'entre et jeton prdit.

Par exemple, si nous avons les phrases:

Alice likes Bob but not cats.

Charlie likes cats.

Nous crons l'ensemble d'entrainement:

Jeton d'entre jeton prdit
Alice likes
Alice Bob
likes Alice
likes Bob
... ...
cats Charlie
cats likes

Implmentez les fonctions suivantes:

  1. __init__

    Compltez le constructeur et construisez l'ensemble d'entrainement comme indiqu plus haut

  1. __getitem__(self, idx)

    Cette fonction retourne l'exemple l'indice voulu idx provenant de l'ensemble d'entrainement. Le format voulu est sous la forme: tuple(jeton d'entre, jeton prdit)

  1. __len__(self)

    Cette fonction retourne la taille de l'ensemble d'entrainement

class SkipGramDataset(Dataset):
    def __init__(self, sequences, window_size=5):
        self.sequences = sequences
        self.window_size = window_size

        self.train_data = []
        for doc in sequences:
            for idx, token in enumerate(doc):
                for i in range(int(-window_size/2), int((window_size/2)+1)):
                    if idx+i < len(doc) and idx+i >= 0 and idx+i != idx:
                        self.train_data.append((token, doc[idx+i]))
                  
                
    def __getitem__(self, idx):
        return self.train_data[idx]
        

    def __len__(self):
        return len(self.train_data)

1.3 Implémentez maintenant l'architecture du Skipgram¶

On vous demande d'implmenter l'architecture du Skipgram avec pytorch. Il vous est aussi demand de faire l'optimisation par batch avec une grandeur de batch variable (dtermin par len(batch_inputs)).

Ci bas se trouve l'architecture gnrale du Skipgram, avec un exemple qui illustre les calculs lors de l'entranement.

Visualisation des matrices de poids:

Visualisation d'un exemple d'entranement. Dans cet exemple, le jeton pass en entre dans le skipgram est "likes":

Exemple d'un batch:

Jeton d'entre jeton prdit
Alice likes
cats Charlie
likes Bob 
# indices :  0        1      2          3        4
vocab =     ["Alice", "Bob", "Charlie", "likes", "cats"]
#              Alice, cats, likes
batch_inputs = [0, 4, 3]

#              likes, Charlie, Bob
batch_labels = [3, 2, 1]

Sois V la taille du vocabulaire et N la taille d'embedding. L'architecture comprend 2 matrices de poids:

  1. La matrice d'encodage $W_i$ de taille $VxN$. Cette matrice est une 'look up table' c'est dire qu'elle prend en entre un index dans le vocabulaire et retourne la ligne de la matrice correspondant cet index. (Correspond torch.nn.Embedding)

  2. La matrice de sortie $W_o$ de taille $NxV$. Cette matrice est multiplie par le vecteur $h$ provenant de l'tape prcdente pour obtenir le vecteur de sortie $y$. (Correspond torch.nn.Linear)

Les fonctions et classes de pytorch dont vous pouvez vous servir sont: torch.nn.Embedding, torch.nn.Linear et torch.nn.log_softmax. Fiez vous la documentation de Pytorch. Voir aussi le tutoriel 3 pour plus d'aide.

Implmentez les fonctions suivantes de la classe SkipGramSoftMax:

  1. __init__

     Compltez le constructeur en initialisant alatoirement les matrices de poids.

  1. forward(self, batch_inputs: List[int])

    param: batch_inputs: List[int]

    return: valeurs softmax avec shape=(len(batch_inputs), len(vocab))

     forward est la mthode de la propagation vers l'avant, qui vous permet de retourner une distribution de probabilit sur le vocabulaire.

 Cette fonction prends en entre un batch de squences,
 puis calcule et retourne les valeurs log_softmax correspondantes.

  1. embed(self, indices)

    param: indices: List[int]

    return: reprsentations vectorielles avec shape=(len(indices), embed_size)

    Cette fonction prend en entre un batch d'indices, et retourne les reprsentations vectorielles (projection/plongements lexicaux). Ceux-ci sont obtenus en calculant le vecteur h dans la figure ci-dessus.

import torch
import torch.nn as nn
from torch.nn import Embedding, Linear, Module
from torch.nn.functional import log_softmax
from typing import List
from sklearn.preprocessing import normalize

device = torch.device("cuda" if torch.cuda.is_available() else "cpu")

class SkipGramSoftMax(Module):
    def __init__(self, vocab, embed_size):
        super().__init__()
        self.embed_size = embed_size
        self.vocab = vocab
        self.embeddings = Embedding(len(vocab), embed_size, device=device)
        self.linear = Linear(embed_size, len(vocab), device=device)
        self.log_softmax = log_softmax
           
        
    def forward(self, batch_inputs: List[int]):
        
        embeds = self.embeddings(torch.tensor(batch_inputs))
        out = self.linear(embeds)
        out = self.log_softmax(out, dim=1)
        return out

    def embed(self, indices: List[int]):

        return self.embeddings(torch.tensor(indices))

1.4 Entrainez le modèle skipgram¶

Compltez le processus d'entranement. Les tapes suivre sont dfinies ci-bas. Voir le tutoriel pour plus d'aide.

  1. Initialiser les gradients zro

  2. Passer les batchs_inputs dans le skipgram afin d'obtenir les valeurs de sortie.

  3. Passer les sorties du rseau et les labels dans la fonction de perte (loss function)

  4. Calculer les valeurs de gradients avec la fonction backward()

  5. Mettre jour les poids avec l'optimiseur. Ici nous utiliserons Adam.

from torch.nn import NLLLoss 
from torch.optim import Adam
from tqdm import tqdm

### Cration du batch loader
skipgram_data = SkipGramDataset(train["sequence"], window_size=5)
batch_loader = DataLoader(skipgram_data, batch_size=128, shuffle=True)

### Cration du modle
device = torch.device("cuda" if torch.cuda.is_available() else "cpu")
skipgram = SkipGramSoftMax(vocab, 100).to(device)

### Fonction permettant de calculer le loss
loss_function = NLLLoss()

### optimiseur suivant la descente de gradiant ADAM
optimizer = Adam(skipgram.parameters())

epochs = 10

for epoch in range(epochs):

    for batch_inputs, batch_labels in tqdm(batch_loader):

        batch_inputs = batch_inputs.to(device)
        batch_labels = batch_labels.to(device)

        ### Initialisation des gradients  0
        optimizer.zero_grad()
        
        ### Calcul de la sortie du modle
        output = skipgram.forward(batch_inputs)
        loss = loss_function(output, batch_labels)
        loss.backward()
        optimizer.step()
    print(f"Epoch {epoch+1} loss: {loss.item()}")

from sklearn.cluster import KMeans

def cluster_and_eval(X_train, y_train, X_test, y_test):
    nbr_clusters = 3
    kmeans = KMeans(n_clusters=nbr_clusters, random_state=0).fit(X_train)
    y_pred = kmeans.predict(X_test)
    repart = np.bincount(y_pred)
    business_index = np.argmax(repart)
    crime_index = np.argmin(repart)
    def f(i):
      if i == business_index:
        return 0
      elif i == crime_index:
        return 2
      else:
        return 1
    
    labels = np.vectorize(f)(y_pred)
    
    print(f"classification report: {classification_report(y_test, labels)}")

    # Pour tiqueter les clusters
    clusters = [np.where(kmeans.labels_ == clus)[0] for clus in range(nbr_clusters)]
    clusters_labels = [y_train[np.where(clus)[0]] for clus in clusters]
    counts = [np.unique(clus_labels, return_counts=True)[1] for clus_labels in clusters_labels]
    for index, number in enumerate(counts):
      print("Cluster {} : il y a {} lments de BUSINESS, COMEDY et CRIME".format(index, number))
